Hashes

empty_hash = Hash.new
=> {}

my_hash = Hash.new("The Default")
my_hash["random_key"]
=> "The Default"

my_hash = {
  "key1" => value1,
  :key2  => value2,
  3 => value 3
}

my_hash = {
  :a => "Artur",
  :l => "Linda",
  :r => "Ryan",
  :z => "Zach"
}
=> {:a => "Artur", :l => "Linda", :r => "Ryan", :z => "Zach"}

SHORTHAND HASHES
my_hash = {
  name: "Artur",
  age:  21
}
=> { :name => "Artur", :age => 21 }
